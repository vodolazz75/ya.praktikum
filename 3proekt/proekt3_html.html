def revenue_from_smth(row, str1, str2, str3, multi):
    revenue = (row[str1] - row[str2]) * row[str3] / multi
    if revenue > 0:
        return revenue
    return 0

def revenue_from_minutes(row):
    return revenue_from_smth(row, 'duration', 'minutes_included', 'rub_per_minute', 1)

def revenue_from_sms(row):
    return revenue_from_smth(row, 'messages', 'messages_included', 'rub_per_message', 1)

def revenue_from_traffic(row):
    return revenue_from_smth(row, 'mb_used', 'mb_per_month_included', 'rub_per_gb', 1024)
